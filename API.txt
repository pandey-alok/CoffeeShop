import requests
import sys
import json

if len(sys.argv)!=3: sys.exit()
else:
    i=sys.argv[2]
    response = requests.get("https://itunes.apple.com/search?entity=song&limit="+i+"&term="+sys.argv[1])
    j=1
    Di=response.json()
    for track in Di["results"]:
        print(j,track["trackName"],"By",track["artistName"])
        j+=1
